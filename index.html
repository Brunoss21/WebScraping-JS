Web Scraping no Navegador com JavaScript

Abra o console (F12) e cole o código abaixo na página da Wikipédia:


document.querySelectorAll('.wikitable a').forEach(link => {
  const span = document.createTextNode(link.textContent);
  link.parentNode.replaceChild(span, link);
});
const linhas = document.querySelectorAll('.wikitable tbody tr');
linhas.forEach((linha, index) => {
  if (index === 0) return;
  const colunas = linha.querySelectorAll('td');
  if (colunas.length < 3) return;
  const ordem = colunas[0]?.textContent.trim() || 'N/A';
  const nome = colunas[1].textContent.trim();
  const mandato = colunas[3].textContent.trim().replace(/\n+/g, ' ');
  const partido = colunas[4].textContent.trim();
  const vicePres = colunas[5].textContent.trim();
  const eleicao = colunas[6].textContent.trim();
  console.log(`Ordem: ${ordem}\nPresidente: ${nome}\nMandato: ${mandato}\nPartido: ${partido}\nVice: ${vicePres}\nEleição: ${eleicao}\n`);
});
